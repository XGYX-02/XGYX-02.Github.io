#include <REGX52.h>  // 包含单片机的寄存器定义

#include "stdio.h"   // 标准输入输出库

#include "string.h"  // 字符串处理库

#include "intrins.h" // 内置函数库



// 定义无符号字符和无符号整型

#define uchar unsigned char 

#define uint unsigned int

#define BT2046_XP 0x9c

#define BT2046_YP 0xDC

#define BT2046_VBAT 0xAC

#define BT2046_AUX 0XEC

// 定义LCD数据寄存器

sfr  lcd_data=0x80; 

// 定义控制引脚

sbit rs=P2^6;  // 数据/指令选择引脚

sbit rw=P2^5;  // 读/写选择引脚

sbit e =P2^7;  // 使能引脚

sbit BT2046_cs=P3^5;

sbit BT2046_dclk=P3^6;

sbit BT2046_din=P3^4;

sbit BT2046_dout=P3^7;

uint ADVALue=0;

// 定义字符集（用于显示的字符）

uchar code table[]={"0123456789abcdef"};



// 延迟函数，控制延迟时间

void yanshi(unsigned int A) //@12.000MHz， 

{

unsigned char data i, j; // 定义局部变量

while(A) // 当A不为0时进入循环

{

i = 2; // 外部循环计数

j = 239; // 内部循环计数

do

{

while (--j); // 内部延迟

} while (--i); // 外部延迟

A--; // 每次循环后A减1

}

}



// 毫秒延迟函数

void mdelay(uint t)

{

uchar i;

while(t--) // 循环t次

{

for(i=0;i<115;i++); // 每次循环产生115次延迟

}

}



// LCD特定延迟函数

void lcd_delay(uint m)

{

while(m--); // 按照输入的m值进行延迟

}



// 向LCD发送命令

void lcd_wcmd(uchar lcdcmd)

{

lcd_delay(60); // 等待60个时钟周期

e=0; // 使能引脚低电平

rs=0; // 选择指令寄存器

rw=0; // 设置写操作

_nop_(); 

lcd_data=lcdcmd; // 将命令写入LCD数据寄存器

_nop_(); // 再次防止编译器优化

_nop_(); // 再次保持时序

e=1; // 使能引脚高电平

_nop_(); // 防止编译器优化

_nop_(); // 再次保持时序

e=0; // 使能引脚恢复为低电平

_nop_(); // 最后一次保持时序

}



// 向LCD写入数据

void lcd_wdata(uchar lcddata)

{

lcd_delay(60); // 等待60个时钟周期

e=0; // 使能引脚低电平

rs=1; // 选择数据寄存器

rw=0; // 设置写操作

_nop_(); // 防止编译器优化

lcd_data=lcddata; // 将数据写入LCD数据寄存器

_nop_(); // 再次防止编译器优化

e=1; // 使能引脚高电平

_nop_(); // 防止编译器优化

_nop_(); // 再次保持时序

e=0; // 使能引脚低电平

_nop_(); // 最后一次保持时序

}



// 初始化LCD

void lcd_init()

{

e=0; // 使能引脚初始化为低电平

lcd_wcmd(0x38); // 设置LCD工作模式

mdelay(1); // 延迟1毫秒

lcd_wcmd(0x08); // 关闭显示

mdelay(2); // 延迟2毫秒

lcd_wcmd(0x01); // 清屏

mdelay(2); // 延迟2毫秒

lcd_wcmd(0x06); // 设定输入模式

mdelay(1); // 延迟1毫秒

lcd_wcmd(0x0c); // 开启显示

mdelay(1); // 延迟1毫秒

}



// 在LCD指定位置显示一个字符

void lcdzf(uchar x, uchar y, uchar shu)

{

y &= 0x01; // 保证y值为0或1

x &= 0x0f; // 限制x值范围

if(y == 1) // 如果y为1，设置为第二行

x |= 0x40; // 移动到第二行地址

x |= 0x80; // 设置特点的LCD地址

lcd_wcmd(x); // 发送地址命令

lcd_wdata(shu); // 发送要显示的字符

}



// 在LCD指定位置显示字符串

void lcdzfc(uchar x, uchar y, uchar code *shu)

{

uchar chang, i;

chang = strlen(shu); // 获取字符串长度

y &= 0x01; // 保证y值为0或1

x &= 0x0f; // 限制x值范围

if(x <= 0x0f) // 检查x值是否有效

{

for(i = 0; i < chang; i++) // 遍历字符串

{

lcdzf(x, y, shu[i]); // 显示每个字符

x++; // 逐个移动到下一个位置

}

}

}

uint BT2046(uchar Command)

{

    uchar i;

    uint ADVALue=0;

    BT2046_dclk=0;

    BT2046_cs=0;



    for (i=0;i<8;i++)

    {

        BT2046_din=Command&0x80;

        BT2046_dclk=1;

        BT2046_dclk=0;




    }

    for(i=0;i<16;i++)

    {

        BT2046_dclk=1;

        BT2046_dclk=0;

        if(BT2046_dout){ADVALue|=(0x8000>>i);}

    }

    BT2046_cs=1;

    return ADVALue>>8;

}



// 主函数

void main()

{

    uint baiwd, shiwd, gewd, wd1, wd2;  // 温度的每一位

    lcd_init();// 初始化LCD

    lcdzfc(0,0,"ADJ");



    while (1)

    {

        ADVALue=BT2046(BT2046_XP);

        baiwd = ADVALue / 10000; // 计算每一位

        shiwd = ADVALue % 10000 / 1000;

        gewd = ADVALue % 10000 % 1000 / 100;

        wd1 = ADVALue % 10000 % 1000 % 100 / 10;

        wd2 = ADVALue % 10000 % 1000 % 100 % 10;

        lcdzf(0, 1, table[baiwd]); // 显示每一位

        lcdzf(1, 1, table[shiwd]);

        lcdzf(2, 1, table[gewd]);

        lcdzf(3, 1, table[wd1]);

        lcdzf(4, 1, table[wd2]);

        yanshi(100);

    }



}
